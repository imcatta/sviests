const _ = require('lodash');
const Round = require('./round.js');
 
class Game {
  constructor(players) {
    this.id = _.uniqueId(Math.floor(Date.now() / 1000));
    this.rounds = [];
    this.newRound(players);
  }
 
  get _whiteCardsUsed() {
    return _.flattenDeep(_.map(this.rounds, (round) => {
      return _.flattenDeep(round.players, (player) => {
        return player.cardIndexes;
      });
    }));
  }
 
  get _blackCardsUsed() {
    return _.flattenDeep(_.map(this.rounds, (round) => {
      return round.blackCard.index;
    }));
  }
 
  get _previousJudges() {
    return _.flattenDeep(_.map(this.rounds, (round) => {
      return round.judgeId;
    }));
  }
 
  get _currentRound() {
    return _.last(this.rounds);
  }
 
  getRoundById(id) {
    return this.rounds.find(r => r.id === id);
  }
 
  newRound(players) {
    const round = new Round(this.id, players, this._blackCardsUsed, this._whiteCardsUsed, this._previousJudges);
    this.rounds.push(round);
    // this._currentRound.allocateCards();
 
    this.whiteCardsUsed = this._whiteCardsUsed;
    this.blackCardsUsed = this._blackCardsUsed;
 
    return round;
  }
}
 
module.exports = Game;
